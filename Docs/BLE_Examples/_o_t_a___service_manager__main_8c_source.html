254 static void OTA_Check_ServiceManager_Operation(void)

301 int main(void)
302 {
303  pFunction Jump_To_Application;
304  uint32_t JumpAddress, appAddress;
305  uint8_t ret;
306 
307  /* Check Service manager RAM Location to verify if a jump to Service Manager has been set from the Application */
308  OTA_Check_ServiceManager_Operation();
309 
310  /* Identifies the valid application where to jump based on the OTA application validity tags values placed on
311  reserved vector table entry: OTA_TAG_VECTOR_TABLE_ENTRY_INDEX */
312  appAddress = OTA_Check_Application_Tags_Value();
313 
314  /* Check if there is a valid application where to jump */
315  if (appAddress == APP_WITH_OTA_SERVICE_ADDRESS)
316  {
317  /* Jump to user application */
318  JumpAddress = *(__IO uint32_t*) (appAddress + 4);
319  Jump_To_Application = (pFunction) JumpAddress;
320  /* Initialize user application's Stack Pointer */
321  __set_MSP(*(__IO uint32_t*) appAddress);
322  Jump_To_Application();
323 
324  /* Infinite loop */
325  while (1)
326  {
327  }
328  }
329 
330  /* Here Ota Service Manager Application is started */
331 
332  /* System Init */
333  SystemInit();
334 
335  /* Identify BlueNRG1 platform */
336  SdkEvalIdentification();
337 
338  SdkEvalComUartInit(UART_BAUDRATE);
339 
340  /* Erase the storage area from start page to end page */
341  OTA_Erase_Flash(APP_WITH_OTA_SERVICE_PAGE_NUMBER_START,APP_WITH_OTA_SERVICE_PAGE_NUMBER_END);
342 
343  /* BlueNRG-1 stack init */
344  ret = BlueNRG_Stack_Initialization(&BlueNRG_Stack_Init_params);
345  if (ret != BLE_STATUS_SUCCESS) {
346  PRINTF("Error in BlueNRG_Stack_Initialization() 0x%02x\r\n", ret);
347  while(1);
348  }
349 
350  PRINTF("\r\nBlueNRG-1 BLE OTA Service Manager (version: %s)\r\n", BLE_OTA_SERVICE_MANAGER_VERSION_STRING);
351 
352  /* Init OTA Service Manager Device */
353  ret = OTA_ServiceManager_DeviceInit();
354  if (ret != BLE_STATUS_SUCCESS) {

356  while(1);
