323 int main(void)
324 {
325  uint8_t charRead, ret;
326 
327  /* System Init */
328  SystemInit();
329 
330  /* Identify BlueNRG1 platform */
331  SdkEvalIdentification();
332 
333  /* Enable the UART */
334  SdkEvalComIOConfig(SdkEvalComIOProcessInputData);
335 
336  /* Disable the PKA peripheral clock to reduce the power consumption */
337  SysCtrl_PeripheralClockCmd(CLOCK_PERIPH_PKA, DISABLE);
338 
339  /* BlueNRG-1 stack init */
340  ret = BlueNRG_Stack_Initialization(&BlueNRG_Stack_Init_params);
341  if (ret != BLE_STATUS_SUCCESS) {
342  printf("Error during BlueNRG_Stack_Initialization() 0x%02x\r\n", ret);
343  while(1);
344  }
345 
346  /* Setup the BLE GAP and GATT layer*/
347  DeviceInit();
348 
349  printf("Power Consumption Test\r\n");
350  printf("Enter ? for list of commands\r\n");
351 
352  device_state = DEVICE_IDLE_STATE;
353  while(1) {
354  BTLE_StackTick();
355 
356  if (__io_getcharNonBlocking(&charRead)) {
357  switch (charRead) {
358  case 'f':
359  if (device_state == DEVICE_IDLE_STATE) {
360  /* Device in discoverable mode with fast adv interval 100 ms */
361  Set_DeviceConnectable(ADV_INTERVAL_FAST_MS);
362  device_state = DEVICE_ADV_STATE;
363  } else {
364  printf ("Device already in discoverable or connected state. Please reset the device.\r\n");
365  }
366  break;
367  case 's':
368  if (device_state == DEVICE_IDLE_STATE) {
369  /* Device in discoverable mode with slow adv interval 1000 ms */
370  Set_DeviceConnectable(ADV_INTERVAL_SLOW_MS);
371  device_state = DEVICE_ADV_STATE;
372  } else {
373  printf ("Device already in discoverable or connected state. Please reset the device.\r\n");
374  }
375  break;
376  case '?':
377  help();
378  break;
379  case 'r':
380  NVIC_SystemReset();
381  break;
382  case '\r':
383  case '\n':
384  // ignored
385  break;
386  default:
387  printf("Unknown Command\r\n");
388  }
389  }
390 
391  switch(device_state) {
392  case DEVICE_DISCONNECTED_STATE:
393  /* Enable the UART */
394  SdkEvalComUartInit(UART_BAUDRATE);
395  printf ("Device Disconnected!!!\r\n");
396  device_state = DEVICE_IDLE_STATE;
397  break;
398  case DEVICE_CONN_STATE:
399  /* Enable the UART */
400  SdkEvalComUartInit(UART_BAUDRATE);
401  printf("Device Connected with Interval %d * 1.25 ms\r\n", connection_interval);
402  device_state = DEVICE_SLEEP_STATE;
403  break;
404  case DEVICE_IDLE_STATE:
405  /* Nothing to Do */
406  break;
407  default:
408  /* Disable the UART to save power */
409  UART_Cmd(DISABLE);
410  SysCtrl_PeripheralClockCmd(CLOCK_PERIPH_UART, DISABLE);
411  SysCtrl_PeripheralClockCmd(CLOCK_PERIPH_GPIO, DISABLE);
412 
413  ret = BlueNRG_Sleep(SLEEPMODE_NOTIMER, 0, 0);
414  if (ret != BLE_STATUS_SUCCESS) {
415  printf("BlueNRG_Sleep() error 0x%02x\r\n", ret);
416  while(1);
417  }
418  }
419  }
420 }
