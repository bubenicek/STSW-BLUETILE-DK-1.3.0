723 uint8_t ADC_SwCalibration(void)
724 {
725  uint32_t ifr0_v1, ifr0_v2, ifr0_v3;
726  float v1f, v2f, v3f;
727 
728  /* Read calibration values */
729  ifr0_v1 = *(uint32_t*)0x100007D0; /* 0.0 V */
730  ifr0_v2 = *(uint32_t*)0x100007D8; /* 1.8 V */
731  ifr0_v3 = *(uint32_t*)0x100007E0; /* 3.6 V */
732 
733  /* Check if the calibration points are valid */
734  if(ifr0_v1 == 0xFFFFFFFF && ifr0_v2 == 0xFFFFFFFF && ifr0_v3 == 0xFFFFFFFF) {
735  return 1;
736  }
737 
738  /* Convert according to the ADC pin 2 formula with PGA 9.54 dB */
739  v1f = (3 * (0.6 - (( ((float)((int16_t)(ifr0_v1>>16))) /ADC_FS_OSR_100_200) * ADC_VREF)));
740  v2f = (3 * (0.6 - (( ((float)((int16_t)(ifr0_v2>>16))) /ADC_FS_OSR_100_200) * ADC_VREF)));
741  v3f = (3 * (0.6 - (( ((float)((int16_t)(ifr0_v3>>16))) /ADC_FS_OSR_100_200) * ADC_VREF)));
742 
743  /* Calculate the slope */
744  slope = 2.0 - ( (v3f - v1f)/3.6 );
745 
746  /* Calculate the slope */
747  offset = (v1f - 1.8 + v2f * slope) / 2.0;
748 
749  return 0;
750 }
