116 void GPIO_Init(GPIO_InitType* GPIO_InitStruct)
117 {
118  uint8_t tmp;
119  uint8_t gpio_mode = 0;
120 
121  /* Check the parameters */
122  assert_param(IS_GPIO_PINS(GPIO_InitStruct->GPIO_Pin));
123  assert_param(IS_GPIO_MODE(GPIO_InitStruct->GPIO_Mode));
124  assert_param(IS_FUNCTIONAL_STATE(GPIO_InitStruct->GPIO_Pull));
125  assert_param(IS_FUNCTIONAL_STATE(GPIO_InitStruct->GPIO_HighPwr));
126 
127  /* IO12 and IO13 cannot be programmed in output mode */
128  assert_param( ((GPIO_InitStruct->GPIO_Pin & GPIO_Pin_12) != GPIO_Pin_12) || (GPIO_InitStruct->GPIO_Mode != GPIO_Output));
129  assert_param( ((GPIO_InitStruct->GPIO_Pin & GPIO_Pin_13) != GPIO_Pin_13) || (GPIO_InitStruct->GPIO_Mode != GPIO_Output));
130 
131  gpio_mode = GPIO_InitStruct->GPIO_Mode;
132 
133  /* Configure the IO mode */
134  if(GPIO_InitStruct->GPIO_Mode == GPIO_Input) {
135  CLEAR_BIT(GPIO->OEN, GPIO_InitStruct->GPIO_Pin);
136  }
137  else if(GPIO_InitStruct->GPIO_Mode == GPIO_Output) {
138  gpio_mode = GPIO_Input;
139  SET_BIT(GPIO->OEN, GPIO_InitStruct->GPIO_Pin);
140  }
141 
142  for(uint8_t i = 0; i < GPIO_PIN_COUNT; i++) {
143  tmp = (uint8_t)((GPIO_InitStruct->GPIO_Pin>>i) & 0x01);
144  if(tmp!=0) {
145 
146  if(i < 8) {
147  MODIFY_REG(GPIO->MODE0, (0xF<<(i*4)), (gpio_mode << (i*4)) );
148  }
149  else if((i > 7) && (i < 16)) {
150  MODIFY_REG(GPIO->MODE1, (0xF<<((i-8)*4)), (gpio_mode << ((i-8)*4)) );
151  }
152 #ifdef BLUENRG2_DEVICE
153  else if((i > 15) && (i < 24)) {
154  MODIFY_REG(GPIO->MODE2, (0xF<<((i-16)*4)), (gpio_mode << ((i-16)*4)) );
155  }
156  else if(i > 23) {
157  MODIFY_REG(GPIO->MODE3, (0xF<<((i-24)*4)), (gpio_mode << ((i-24)*4)) );
158  }
159 #endif
160  }
161  }
162 
163  /* Set the current mode */
164  if(GPIO_InitStruct->GPIO_HighPwr == ENABLE) {
165  SET_BIT(GPIO->DS, GPIO_InitStruct->GPIO_Pin);
166  }
167  else {
168  CLEAR_BIT(GPIO->DS, GPIO_InitStruct->GPIO_Pin);
169  }
170 
171  /* Set the pull on IO */
172  if(GPIO_InitStruct->GPIO_Pull == ENABLE) {
173  SET_BIT(GPIO->PE, GPIO_InitStruct->GPIO_Pin);
174  }
175  else {
176  CLEAR_BIT(GPIO->PE, GPIO_InitStruct->GPIO_Pin);
177  }
178 
179 }

413 void GPIO_EXTIConfig(GPIO_EXTIConfigType* EXTIConfig)
414 {
415  /* Check the parameters */
416  assert_param(IS_GPIO_PINS(EXTIConfig->GPIO_Pin));
417  assert_param(IS_GPIO_IRQSENSE(EXTIConfig->GPIO_IrqSense));
418  assert_param(IS_GPIO_EVENT(EXTIConfig->GPIO_Event));
419 
420  /* Interrupt sense configuration */
421  if (EXTIConfig->GPIO_IrqSense == GPIO_IrqSense_Level) {
422  /* Configure level detection */
423  SET_BIT(GPIO->IS, EXTIConfig->GPIO_Pin);
424  }
425  else {
426  /* Configure edge detection */
427  CLEAR_BIT(GPIO->IS, EXTIConfig->GPIO_Pin);
428  }
429 
430  /* Configure event */
431  if (EXTIConfig->GPIO_Event == GPIO_Event_High) {
432  /* Configure interrupt on high level or rising edge event */
433  SET_BIT(GPIO->IEV, EXTIConfig->GPIO_Pin);
434  CLEAR_BIT(GPIO->IBE, EXTIConfig->GPIO_Pin);
435  }
436  else if (EXTIConfig->GPIO_Event == GPIO_Event_Low) {
437  /* Configure interrupt on low level or falling edge event */
438  CLEAR_BIT(GPIO->IEV, EXTIConfig->GPIO_Pin);
439  CLEAR_BIT(GPIO->IBE, EXTIConfig->GPIO_Pin);
440  }
441  else {
442  /* Configure interrupt on both edge */
443  SET_BIT(GPIO->IBE, EXTIConfig->GPIO_Pin);
444  }
445 
446 }
