192 struct ActionPacket
193 {
194  uint8_t StateMachineNo ; /* This parameter indicates the state machine number for this action. From 0 to 7. */
195  uint8_t ActionTag; /* The configuration of the current action.
196  * Action Tag: PLL_TRIG, TXRX, TIMER_WAKEUP, INC_CHAN, TIMESTAMP_POSITION, RELATIVE */
197  uint32_t WakeupTime; /* Contains the wakeup time in microsecond if it is relative.
198  * It should not be more than 24 bits if it is absolute.
199  * It only applies if TIMER_WAKEUP flag is set in ActionTag. */
200  uint32_t ReceiveWindowLength; /* Sets RX window size in microsecond. Applicable only for RX actions. */
201  uint8_t *data; /* Pointer to the array with the data to send (header, length and data field), for TX.
202  * Pointer to the array where the data received are copied, for RX.
203  * In case of RX, the array must have the max size MAX_PACKET_LENGTH. */
204  uint32_t status; /* The Status Register with the information on the action. */
205  uint32_t timestamp_receive; /* This field contains the timestamp when a packet is received.
206  * Intended to be used in the dataRoutine() callback routine. RX only. */
207  int32_t rssi; /* The rssi of the packet was received with. RX only. */
208  BlueTransStruct trans_packet; /* This is a linked list, which is going to be used by Hardware.
209  * User does not need to do anything. */
210  ActionPacket *next_true; /* Pointer to next ActionPacket if condRoutine() returns TRUE */
211  ActionPacket *next_false; /* Pointer to next ActionPacket if condRoutine() returns FALSE */
212  uint8_t (*condRoutine)(ActionPacket*); /* User callback that decide the next ActionPacket to use.
213  * It is time critical. Routine must end within 45 us. */
214  uint8_t (*dataRoutine)(ActionPacket*, ActionPacket*); /* User callback for managing data. */
215  uint8_t trans_config; /* This is for configuring the device for TX or RX. User does not need to do anything. */
216 };
