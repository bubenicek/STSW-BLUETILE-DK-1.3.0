141 void UART_Init(UART_InitType* UART_InitStruct)
142 {
143  uint32_t divider;
144  uint16_t ibrd, fbrd;
145 
146  /* Check the parameters */
147  assert_param(IS_UART_BAUDRATE(UART_InitStruct->UART_BaudRate));
148  assert_param(IS_UART_WORD_LENGTH(UART_InitStruct->UART_WordLengthTransmit));
149  assert_param(IS_UART_WORD_LENGTH(UART_InitStruct->UART_WordLengthReceive));
150  assert_param(IS_UART_STOPBITS(UART_InitStruct->UART_StopBits));
151  assert_param(IS_UART_PARITY(UART_InitStruct->UART_Parity));
152  assert_param(IS_UART_MODE(UART_InitStruct->UART_Mode));
153  assert_param(IS_UART_HW_FLOW_CONTROL(UART_InitStruct->UART_HardwareFlowControl));
154  assert_param(IS_FUNCTIONAL_STATE(UART_InitStruct->UART_FifoEnable));
155 
156  /*---------------------------- UART BaudRate Configuration -----------------------*/
157 
158  if (UART->CR_b.OVSFACT == 0)
159  divider = (UART_CLOCK<<7) / (UART_CLOCK_CYCLE16 * UART_InitStruct->UART_BaudRate);
160  else
161  divider = (UART_CLOCK<<7) / (UART_CLOCK_CYCLE8 * UART_InitStruct->UART_BaudRate);
162 
163  ibrd = divider >> 7;
164  UART->IBRD = ibrd;
165  fbrd = ((divider - (ibrd <<7) + 1) >> 1);
166  if (fbrd > 0x3f) {
167  ibrd++;
168  fbrd = (fbrd - 0x3F) & 0x3F;
169  }
170  UART->FBRD = fbrd;
171 
172  /*---------------------------- UART Word Length Configuration -----------------------*/
173  UART->LCRH_TX_b.WLEN_TX = UART_InitStruct->UART_WordLengthTransmit;
174  UART->LCRH_RX_b.WLEN_RX = UART_InitStruct->UART_WordLengthReceive;
175 
176  /*---------------------------- UART Stop Bits Configuration -----------------------*/
177  UART->LCRH_TX_b.STP2_TX = UART_InitStruct->UART_StopBits;
178  UART->LCRH_RX_b.STP2_RX = UART_InitStruct->UART_StopBits;
179 
180  /*---------------------------- UART Parity Configuration -----------------------*/
181  if(UART_InitStruct->UART_Parity == UART_Parity_No) {
182  UART->LCRH_TX_b.PEN_TX = RESET;
183  UART->LCRH_RX_b.PEN_RX = RESET;
184  }
185  else if(UART_InitStruct->UART_Parity == UART_Parity_Odd) {
186  UART->LCRH_TX_b.PEN_TX = SET;
187  UART->LCRH_TX_b.EPS_TX = RESET;
188  UART->LCRH_RX_b.PEN_RX = SET;
189  UART->LCRH_RX_b.EPS_RX = RESET;
190  }
191  else {
192  UART->LCRH_TX_b.PEN_TX = SET;
193  UART->LCRH_TX_b.EPS_TX = SET;
194  UART->LCRH_RX_b.PEN_RX = SET;
195  UART->LCRH_RX_b.EPS_RX = SET;
196  }
197 
198  /*---------------------------- UART Mode Configuration -----------------------*/
199  UART->CR_b.RXE = UART_InitStruct->UART_Mode&1;
200  UART->CR_b.TXE = (UART_InitStruct->UART_Mode>>1)&1;
201 
202  /*---------------------------- UART Hardware flow control Configuration -----------------------*/
203  UART->CR_b.RTSEN = UART_InitStruct->UART_HardwareFlowControl&1;
204  UART->CR_b.CTSEN = (UART_InitStruct->UART_HardwareFlowControl>>1)&1;
205 
206  /*---------------------------- UART Fifo Configuration -----------------------*/
207  UART->LCRH_TX_b.FEN_TX = (uint8_t)UART_InitStruct->UART_FifoEnable;
208  UART->LCRH_RX_b.FEN_RX = (uint8_t)UART_InitStruct->UART_FifoEnable;
209 
210 }

486 void UART_RXTimeoutConfig(uint32_t UART_TimeoutMS)
487 {
488  uint32_t timeout_val;
489  uint32_t tmp32;
490 
491  tmp32 = (UART->IBRD)<<4;
492  if (UART->CR_b.OVSFACT == 0)
493  timeout_val = (UART_TimeoutMS * 16 * 16000) / (((UART->FBRD)>>2) + tmp32);
494  else
495  timeout_val = (UART_TimeoutMS * 8 * 16000) / (((UART->FBRD)>>3) + tmp32);
496 
497  /* Set the timeout value */
498  if(IS_UART_TIMEOUT(timeout_val)) {
499  UART->TIMEOUT = timeout_val;
500  }
501  else {
502  UART->TIMEOUT = UART_MAX_RX_TIMEOUT;
