426 void TC_IT_Callback(void)
427 {
428  uint16_t nBytes, nPCMs = 0;
429 
430  for ( nBytes = 0; nBytes< (FS/1000)*2; nBytes++) /* FS/1000 PCM samples(16 bit) -> (FS/1000)*2 in Bytes */
431  {
432  if((nBytes%2)==0) {
433  HP_Filter.Z = PCM_Buffer[nPCMs + PCM_BUFFER_SIZE/2] * AudioVolume;
434  HP_Filter.oldOut = (0xFC * (HP_Filter.oldOut + HP_Filter.Z - HP_Filter.oldIn)) / 256;
435  HP_Filter.oldIn = HP_Filter.Z;
436  PCM_Buffer[nPCMs + PCM_BUFFER_SIZE/2] = SaturaLH(HP_Filter.oldOut, -32760, 32760);
437  nPCMs++;
438  }
439  /* Send the PDM data through UART */
440  while (UART_GetFlagStatus(UART_FLAG_TXFE) == RESET);
441  UART_SendData( ((uint8_t *)PCM_Buffer)[nBytes + (FS/1000)*2]); /* FS/1000 PCM samples(16 bit) -> (FS/1000)*2 in Bytes */
442  }
443 
444 }
