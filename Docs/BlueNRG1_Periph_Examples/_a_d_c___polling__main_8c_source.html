255  if(ADC_SwCalibration())
256  printf("No calibration points found. SW compensation cannot be done.\r\n");

259  ADC_CONFIGURATION();
260 
261  /* Start new conversion */
262  ADC_ENABLE();
263 
264  /* Infinite loop */
265  while(1) {
266 
267  /* Polling of End-Of-Conversion flag */
268  if(ADC_CHECK_FLAG) {
269

271  adc_value = ADC_GetConvertedData(xADC_InitType.ADC_Input, xADC_InitType.ADC_ReferenceVoltage);
272 
273  /* Print the ADC value converted */
274  if(xADC_InitType.ADC_Input == ADC_Input_TempSensor) {
275  printf("ADC value: %d.%02d %cC\r\n", PRINT_INT(adc_value),PRINT_FLOAT(adc_value), 248);
276  }
277  else {
278  printf("ADC value: %d.%03d mV\r\n", PRINT_INT(ADC_CompensateOutputValue(adc_value)*1000.0), PRINT_FLOAT(ADC_CompensateOutputValue(adc_value)*1000.0));
279  }
280  SdkEvalLedToggle(LED1);
281 
282  /* Application delay before next one shot measurement */
283  SdkDelayMs(100);
284 
285  /* Start new conversion */
286  ADC_ENABLE();
287  }
