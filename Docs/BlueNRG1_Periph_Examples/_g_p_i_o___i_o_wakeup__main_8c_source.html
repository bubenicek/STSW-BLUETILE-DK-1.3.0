223  GPIO_InitType GPIO_InitStructure;
224  GPIO_InitStructure.GPIO_Pin = Get_LedGpioPin(LED1) | Get_LedGpioPin(LED2);
225  GPIO_InitStructure.GPIO_Mode = GPIO_Output;
226  GPIO_InitStructure.GPIO_Pull = ENABLE;
227  GPIO_InitStructure.GPIO_HighPwr = ENABLE;

234  GPIO_InitStructure.GPIO_Pin = Get_ButtonGpioPin(USER_BUTTON);
235  GPIO_InitStructure.GPIO_Mode = GPIO_Input;
236  GPIO_InitStructure.GPIO_Pull = DISABLE;
237  GPIO_InitStructure.GPIO_HighPwr = DISABLE;
