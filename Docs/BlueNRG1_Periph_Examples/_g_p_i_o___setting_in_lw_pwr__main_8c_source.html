225 {
226  uint8_t ret, wakeup_source, wakeup_level;
227 
228  wakeup_source = WAKEUP_IO13;
229  wakeup_level = (WAKEUP_IOx_LOW << WAKEUP_IO13_SHIFT_MASK);
230 
231  printf("\r\nEnter to Standby - SLEEPMODE_NOTIMER\r\n");
232  while(SdkEvalComUARTBusy() == SET);
233  ret = BlueNRG_Sleep(SLEEPMODE_NOTIMER, wakeup_source, wakeup_level);
234  if (ret != BLE_STATUS_SUCCESS) {
235  printf("BlueNRG_Sleep() error 0x%02x\r\n", ret);
236  while(1);
237  }
238  printf("Wake-up!!\r\n");

242 int main(void)
243 {
244  uint8_t charRead, ret;

246  /* System Init */
247  SystemInit();
248 
249  /* Identify BlueNRG1 platform */
250  SdkEvalIdentification();

254 
255  /* Configure the BUTTON_1 button */
256  SdkEvalPushButtonInit(BUTTON_1);
257  /* IRQ_ON_FALLING_EDGE is set since wakeup level is low (mandatory setup for wakeup sources);
258  IRQ_ON_RISING_EDGE is the specific application level */
259  SdkEvalPushButtonIrq(BUTTON_1, IRQ_ON_BOTH_EDGE);
260 
261  /* Led DL1 configuration */
262  SdkEvalLedInit(LED1);
263  SdkEvalLedOn(LED1);
264 
265  /* GPIO setting in low power mode: output */
266  GPIO_InitType GPIO_InitStructure;
267 
268  /* Configures Button pin as input */
269  GPIO_InitStructure.GPIO_Pin = GPIO_Pin_9 | GPIO_Pin_10 | GPIO_Pin_11;
270  GPIO_InitStructure.GPIO_Mode = GPIO_Output;
271  GPIO_InitStructure.GPIO_Pull = DISABLE;
272  GPIO_InitStructure.GPIO_HighPwr = DISABLE;
273 
274  /* GPIO9, GPIO10, GPIO11 initialization during low power modes for BlueNRG-2. */
275  GPIO_InitLowPowerModes(&GPIO_InitStructure);
276  /* Set the output state of the GIO9, GIO10, GIO11 during low power modes. BlueNRG-2 only */
277  GPIO_WriteBitsLowPowerModes(GPIO_Pin_9 | GPIO_Pin_10 | GPIO_Pin_11, Bit_RESET);
278 
279  /* BlufeNRG-1 stack init */
280  ret = BlueNRG_Stack_Initialization(&BlueNRG_Stack_Init_params);
281  if (ret != BLE_STATUS_SUCCESS) {
282  printf("Error during BlueNRG_Stack_Initialization() 0x%02x\r\n", ret);
283  while(1);
284  }
285 
286  printf("========================================================\r\n");
287  printf("GPIOs setting during low power mode.\r\n");
288  printf("The GPIO9, GPIO10 and GPIO11 can be set\r\n to retains their output state during low power mode.\r\n");
289  printf("See the STEVAL-IDB008Vx schematic for the location of these GPIOs.\r\n");
290  printf("Note that the GPIO9 and GPIO10 are the SWD pins.\r\n");
291  printf("========================================================\r\n");
292  printf("Enter ? for list of commands\r\n");
293 
294  printf("SYSTEM_CTRL->SLEEPIO_OEN %08X\r\n", SYSTEM_CTRL->SLEEPIO_OEN);
295  printf("CKGEN_SOC->CLOCK_EN %08X\r\n", CKGEN_SOC->CLOCK_EN);
296  while(1) {
297  BTLE_StackTick();
298 
299  if (__io_getcharNonBlocking(&charRead)) {

302  printf("Press PUSH1 button to wake-up the BlueNRG-2.\r\n");
303  sleep();
304  break;
305  case '1':
306  printf("Toggle GPIO9 low power output state\r\n");
307  GPIO_ToggleBitsLowPowerModes(GPIO_Pin_9);
308  printf("SYSTEM_CTRL->SLEEPIO_OUT %08X\r\n", SYSTEM_CTRL->SLEEPIO_OUT);

311  printf("Toggle GPIO10 low power output state\r\n");
312  GPIO_ToggleBitsLowPowerModes(GPIO_Pin_10);

314  break;
315  case '3':
316  printf("Toggle GPIO11 low power output state\r\n");
317  GPIO_ToggleBitsLowPowerModes(GPIO_Pin_11);
318  printf("SYSTEM_CTRL->SLEEPIO_OUT %08X\r\n", SYSTEM_CTRL->SLEEPIO_OUT);
319  break;
320  case '?':
321  help();
322  break;
323  case 'r':
324  NVIC_SystemReset();
325  break;
326  case '\r':
327  case '\n':
328  // ignored
329  break;

331  printf("Unknown Command\r\n");
332  }
333  }
334  }
