308 void processCommand(void)
309 {
310  uint8_t command;
311  uint8_t led_status;
312 
313  /* Infinite loop */
314  while(1) {
315 
316  /* Loop until the UART Receive Data Register is not empty */
317  while (UART_GetFlagStatus(UART_FLAG_RXFE) == SET);
318 
319  /* Store the received byte in RxBuffer */
320  command = UART_ReceiveData();
321 
322  /* Parse the command */
323  switch (command) {
324 
325  case TOGGLE_DL1:
326  /* I2C Master communication */
327  I2C_Read_Reg(DEV_ADDRESS, TOGGLE_DL1, &led_status);
328 
329  if((led_status&1) == 1) {
330  printf("LED DL1 OFF\r\n");
331  }
332  else {
333  printf("LED DL1 ON\r\n");
334  }
335  break;
336 
337  case TOGGLE_DL2:
338  /* I2C Master communication */
339  I2C_Read_Reg(DEV_ADDRESS, TOGGLE_DL2, &led_status);
340 
341  if((led_status&2) == 2) {
342  printf("LED DL2 OFF\r\n");
343  }
344  else {
345  printf("LED DL2 ON\r\n");
346  }
347  break;
348 
349  case TOGGLE_DL3:
350  /* I2C Master communication */
351  I2C_Read_Reg(DEV_ADDRESS, TOGGLE_DL3, &led_status);
352 
353  if((led_status&4) == 4) {
354  printf("LED DL3 OFF\r\n");
355  }
356  else {
357  printf("LED DL3 ON\r\n");
358  }
359  break;
360 
361  default:
362  helpMessage();
363  break;
364  }
365  led_status = 0;
366  }

416 I2C_AppOpStatus I2C_Read_Reg(uint8_t dev_address, uint8_t send_val, uint8_t* get_val)
417 {
418  I2C_TransactionType t;
419 
420  /* Write I2C device address address */
421  t.Operation = I2C_Operation_Write;
422  t.Address = dev_address;
423  t.StartByte = I2C_StartByte_Disable;
424  t.AddressType = I2C_AddressType_7Bit;
425  t.StopCondition = I2C_StopCondition_Disable;
426  t.Length = 1;
427 
428  I2C_FlushTx((I2C_Type*)SDK_EVAL_I2C);
429 
430  /* wait until Flush finished */
431  while (I2C_WaitFlushTx((I2C_Type*)SDK_EVAL_I2C) == I2C_OP_ONGOING);
432 
433  /* Write I2C device address address and put the send_val in TX FIFO */
434  I2C_BeginTransaction((I2C_Type*)SDK_EVAL_I2C, &t);
435  I2C_FillTxFIFO((I2C_Type*)SDK_EVAL_I2C, send_val);
436 
437  /* Check write */
438  do
439  {
440  if(I2C_OP_ABORTED == I2C_GetStatus((I2C_Type*)SDK_EVAL_I2C))
441  return I2C_OP_FAILED;
442 
443  } while (I2C_GetITStatus((I2C_Type*)SDK_EVAL_I2C, I2C_IT_MTDWS) == RESET);
444 
445  I2C_ClearITPendingBit((I2C_Type*)SDK_EVAL_I2C, I2C_IT_MTDWS);
446 
447  /* Prepare to read data */
448  t.Operation = I2C_Operation_Read;
449  t.Address = dev_address;
450  t.StartByte = I2C_StartByte_Disable;
451  t.AddressType = I2C_AddressType_7Bit;
452  t.StopCondition = I2C_StopCondition_Enable;
453  t.Length = 1;
454 
455  I2C_BeginTransaction((I2C_Type*)SDK_EVAL_I2C, &t);
456 
457  /* Check read */
458  do
459  {
460  if(I2C_OP_ABORTED == I2C_GetStatus((I2C_Type*)SDK_EVAL_I2C))
461  return I2C_OP_FAILED;
462 
463  } while (RESET == I2C_GetITStatus((I2C_Type*)SDK_EVAL_I2C, I2C_IT_MTD));
464 
465  I2C_ClearITPendingBit((I2C_Type*)SDK_EVAL_I2C, I2C_IT_MTD | I2C_IT_MTDWS);
466 
467  /* Get data */
468  *get_val = I2C_ReceiveData((I2C_Type*)SDK_EVAL_I2C);
469 
470  return I2C_OP_SUCCEED;
471 }
