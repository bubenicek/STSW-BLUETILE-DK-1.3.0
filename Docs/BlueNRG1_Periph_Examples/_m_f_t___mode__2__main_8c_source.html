236  /* IO3 for TnCRB (TnB) */
237  MFT_SelectCapturePin(MFT1_TIMERB, IO_CAPTURE_PIN_MFT1_TIMERB);
238 
239  /* MFT configuration */
240  MFT_Configuration();
241 
242  /* Configure edge detection on falling edge for TnA and TnB */
243  MFT_TnEDGES(MFT1, MFT_FALLING, MFT_FALLING);
244 
245  /* Disable the preset (to 0xFFFF) of the TnCNT1 register on TnA and TnB event.
246  * In this way, the TnCNT1 value is not changed by hardware.
247  * This allows to use its underflow in order to compute the frequency of
248  * signals greater than the max tick counter. */
249  MFT_TnXEN(MFT1, MFT_TnA, DISABLE);
250  MFT_TnXEN(MFT1, MFT_TnB, DISABLE);
251 
252  /* Enable all MFT interrupt */
253  MFT_EnableIT(MFT1, MFT_IT_TNA | MFT_IT_TNB | MFT_IT_TNC, ENABLE);
254 
255  /* Start MFT1 timer */
256  MFT_Cmd(MFT1, ENABLE);
257 
258  /* Infinite loop */
259  while(1) {
260 
261  /* If input signal frequency on IN_PIN1 is 1kHz OUT_PIN2 is in LOW state */
262  if (fA_found == SET) {
263  SdkEvalLedOn(LED1);
264  }
265  else {
266  SdkEvalLedOff(LED1);
267  }
268 
269  /* If input signal frequency on IN_PIN2 is 10kHz OUT_PIN3 is in LOW state */
270  if (fB_found == SET) {
271  SdkEvalLedOn(LED2);
272  }

287 void MFT_Configuration()
288 {
289  MFT_InitType timer_init;
290  NVIC_InitType NVIC_InitStructure;
291 
292  SysCtrl_PeripheralClockCmd(CLOCK_PERIPH_MTFX1, ENABLE);
293 
294  /* Init MFT1 in mode 2 */
295  timer_init.MFT_Clock1 = MFT_PRESCALED_CLK;
296  timer_init.MFT_Clock2 = MFT_NO_CLK;
297  timer_init.MFT_Mode = MFT_MODE_2;
298  timer_init.MFT_CRA = 0xFFFF;
299  timer_init.MFT_CRB = 0xFFFF;
300 
301  /* If a square waveform is @ input with a frequency of 10kHz => T = 100 us
302  *
303  * with Prescaler = 16 - 1; => 1us sampling period
304  * the counter number within a period is 100.
305  *
306  * with Prescaler = 1 - 1; => 62.5ns sampling period
307  * the counter number within a period is 16000. More resolution.
308  */
309  timer_init.MFT_Prescaler = 1-1;
310 
311  MFT_Init(MFT1, &timer_init);
312 
313  /* Set the TnCNT1 at the max value. No matter about the TnCNT2 since it is not used. */
314  MFT_SetCounter(MFT1, 0xFFFF, 0xFFFF);
315 
316  /* Enable MFT irq */
317  NVIC_InitStructure.NVIC_IRQChannel = MFT1A_IRQn;
318  NVIC_InitStructure.NVIC_IRQChannelPreemptionPriority = LOW_PRIORITY;
319  NVIC_InitStructure.NVIC_IRQChannelCmd = ENABLE;
320  NVIC_Init(&NVIC_InitStructure);
321 }
