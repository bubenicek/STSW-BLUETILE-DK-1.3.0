271 void MFT_Configuration()
272 {
273  MFT_InitType timer_init;
274  NVIC_InitType NVIC_InitStructure;
275 
276  SysCtrl_PeripheralClockCmd(CLOCK_PERIPH_MTFX1, ENABLE);
277 
278  /* Init MFT3 in mode 3 */
279  /* Clock 1 is from prescaled clock */
280  /* Clock 2 is external clock */
281  /* Counter 1 is preset and reloaded to 5000 (timer period = (TnCRA * 2) * Prescaler/System_Clock = 10 ms if System_Clock is 16 MHz) */
282  /* Counter 2 is preset and reloaded to 5 (timer period = TnCRB * 2 * 10 ms = 100 ms */
283  timer_init.MFT_Mode = MFT_MODE_3;
284  timer_init.MFT_Clock1 = MFT_PRESCALED_CLK;
285  timer_init.MFT_Prescaler = 16 - 1; /* 1 us tick at 16 MHz system clock */
286  timer_init.MFT_Clock2 = MFT_EXTERNAL_EVENT;
287  timer_init.MFT_CRA = 5000 - 1;
288  timer_init.MFT_CRB = 5 - 1;
289  MFT_Init(MFT1, &timer_init);
290 
291  /* Set counter for timer1 and timer2 */
292  MFT_SetCounter(MFT1, 5000 - 1, 5 - 1);
293 
294  /* Enable MFT Interrupts */
295  NVIC_InitStructure.NVIC_IRQChannel = MFT1B_IRQn;
296  NVIC_InitStructure.NVIC_IRQChannelPreemptionPriority = LOW_PRIORITY;
297  NVIC_InitStructure.NVIC_IRQChannelCmd = ENABLE;
298  NVIC_Init(&NVIC_InitStructure);
299 }
