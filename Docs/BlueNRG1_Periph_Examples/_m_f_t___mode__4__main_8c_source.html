292  timer_init.MFT_Clock2 = MFT_PRESCALED_CLK;

294  timer_init.MFT_Prescaler = 16 - 1; /* 1 us if system_clock is 16 MHz */
295 
296  timer_init.MFT_CRA = 50000-1; /* 50 ms */
297  timer_init.MFT_CRB = 0xFFFF;
298
