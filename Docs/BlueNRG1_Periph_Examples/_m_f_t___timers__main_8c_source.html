234 void MFT_Configuration(void)
235 {
236  NVIC_InitType NVIC_InitStructure;
237  MFT_InitType timer_init;
238 
239  SysCtrl_PeripheralClockCmd(CLOCK_PERIPH_MTFX1 | CLOCK_PERIPH_MTFX2, ENABLE);
240 
241  MFT_StructInit(&timer_init);
242 
243  timer_init.MFT_Mode = MFT_MODE_1;
244  timer_init.MFT_Prescaler = 160-1; /* 10 us clock */
245 
246  /* MFT1 configuration */
247  timer_init.MFT_Clock2 = MFT_PRESCALED_CLK;
248  MFT_Init(MFT1, &timer_init);
249  /* Set the counter at 500 ms */
250  MFT_SetCounter2(MFT1, 50000);
251 
252  /* MFT2 configuration */
253  timer_init.MFT_Clock2 = MFT_PRESCALED_CLK;
254  MFT_Init(MFT2, &timer_init);
255  /* Set the counter at 250 ms */
256  MFT_SetCounter2(MFT2, 25000);
257 
258  /* Enable MFT2 Interrupt 1 */
259  NVIC_InitStructure.NVIC_IRQChannel = MFT1B_IRQn;
260  NVIC_InitStructure.NVIC_IRQChannelPreemptionPriority = LOW_PRIORITY;
261  NVIC_InitStructure.NVIC_IRQChannelCmd = ENABLE;
262  NVIC_Init(&NVIC_InitStructure);
263 
264  /* Enable MFT2 Interrupt 2 */
265  NVIC_InitStructure.NVIC_IRQChannel = MFT2B_IRQn;
266  NVIC_Init(&NVIC_InitStructure);
267 
269  MFT_EnableIT(MFT1, MFT_IT_TND, ENABLE);
270  MFT_EnableIT(MFT2, MFT_IT_TND, ENABLE);
271 }
