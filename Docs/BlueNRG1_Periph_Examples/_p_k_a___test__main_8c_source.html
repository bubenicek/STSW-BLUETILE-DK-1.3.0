199 void GenerateRandomK16Word(uint32_t* buffer)
200 {
201  SysCtrl_PeripheralClockCmd(CLOCK_PERIPH_RNG, ENABLE);
202 
203  /* Create a random key */
204  for(uint8_t i=0; i<16; i++) {
205 
206  /* Loop until the RNG Data Ready is SET */
207  while (RNG_GetFlagStatus() != SET);
208 
209  /* Generate a Random Number and load it into ecc_addr_k register */
210  buffer[i] = RNG_GetValue();
211 
212  /* Loop until the RNG Data Ready is SET */
213  while (RNG_GetFlagStatus() != SET);
214 
215  /* Generate a Random Number and load it into ecc_addr_k register */
216  buffer[i] |= (((uint32_t)RNG_GetValue())<<16);
217  }
218 
219  SysCtrl_PeripheralClockCmd(CLOCK_PERIPH_RNG, DISABLE);
220 }
