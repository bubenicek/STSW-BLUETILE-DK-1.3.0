263 uint8_t dataRoutine(ActionPacket* p, ActionPacket* next)
264 {
265  /* received a packet */
266  if( (p->status & BIT_TX_MODE) == 0) {
267 
268  if((p->status & IRQ_RCV_OK) != 0) {
269  // if(ack_counter_test==0)
270  // pkt_counter_test = 0;
271 
272  if(p->data[0] == 0xAE) {
273  ack_counter_test[comm_token-1]++;
274  pkt_counter_test[comm_token-1]++;
275  retry_counter_test = 0;
276  repeat_flag = SCHEDULE_NEW_PACKET;
277  }
278  else {
279  retry_counter_test++;
280  retry_counter_test_max[comm_token-1]++;
281  repeat_flag = SCHEDEULE_RETRY;
282  }
283  }
284  else if((p->status & IRQ_TIMEOUT) != 0) {
285  retry_counter_test++;
286  retry_counter_test_max[comm_token-1]++;
287  repeat_flag = SCHEDEULE_RETRY;
288  }
289  else if((p->status & IRQ_CRC_ERR) != 0) {
290  retry_counter_test++;
291  retry_counter_test_max[comm_token-1]++;
292  repeat_flag = SCHEDEULE_RETRY;
293  }
294 
295  if( retry_counter_test == MAX_RETRY_TEST) {
296  pkt_counter_test[comm_token-1]++;
297  retry_counter_test = 0;
298  repeat_flag = SCHEDULE_NEW_PACKET;
299  }
300 
301  }
302  /* Transmit complete */
303  else {
304  }
305  return TRUE;
306 }
