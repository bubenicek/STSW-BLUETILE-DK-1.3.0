251 uint8_t RxCallback(ActionPacket* p, ActionPacket* next)
252 {
253  /* received a packet */
254  if( (p->status & BIT_TX_MODE) == 0) {
255 
256  if((p->status & IRQ_RCV_OK) != 0) {
257  if(LEDisON) {
258  SdkEvalLedOff(LED1);
259  LEDisON = 0;
260  }
261  else {
262  SdkEvalLedOn(LED1);
263  LEDisON = 1;
264  }
265 
266  if(flag_SendingPacket == FALSE) {
267  HAL_RADIO_ReceivePacket(FREQUENCY_CHANNEL, RX_RELATIVETIME, receivedData, RX_RECEIVE_TIMEOUT, RxCallback);
268  }
269  else {
270  HAL_RADIO_SendPacket(FREQUENCY_CHANNEL, TX_RELATIVETIME, sendData, TxCallback);
271  flag_SendingPacket = FALSE;
272  }
273  }
274  else if(((p->status & IRQ_TIMEOUT) != 0) || ((p->status & IRQ_CRC_ERR) != 0)) {
275  if(flag_SendingPacket == FALSE) {
276  HAL_RADIO_ReceivePacket(FREQUENCY_CHANNEL, RX_RELATIVETIME, receivedData, RX_RECEIVE_TIMEOUT, RxCallback);
277  }
278  else {
279  HAL_RADIO_SendPacket(FREQUENCY_CHANNEL, TX_RELATIVETIME, sendData, TxCallback);
280  flag_SendingPacket = FALSE;
281  }
282  }
283  }
284  return TRUE;
285 }
