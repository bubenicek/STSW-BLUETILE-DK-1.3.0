343 void InitializationActionPackets(void)
344 {
345  /* Build Action Packets */
346  waitRequestAction.StateMachineNo = STATE_MACHINE_0;
347  waitRequestAction.ActionTag = RELATIVE | TIMER_WAKEUP | PLL_TRIG;
348  waitRequestAction.WakeupTime = 30+WKP_OFFSET; /* 30 us minimum of internal delay before start action */
349  waitRequestAction.ReceiveWindowLength = RX_TIMEOUT_DATA; /* Not applied for TX */
350  waitRequestAction.data = receiveBuffer; /* Data to send */
351  waitRequestAction.next_true = &sendDataAction; /* If condRoutine returns TRUE => Go to next TX the 2nd part of the payload */
352  waitRequestAction.next_false = &waitRequestAction; /* Not Used */
353  waitRequestAction.condRoutine = waitRequestCB; /* Condition routine */
354  waitRequestAction.dataRoutine = dataRoutine; /* Data routine */
355 
356  sendDataAction.StateMachineNo = STATE_MACHINE_0;
357  sendDataAction.ActionTag = RELATIVE | TIMER_WAKEUP | TXRX;
358  sendDataAction.WakeupTime = 40+WKP_OFFSET; /* 40 us minimum of internal delay before start action */
359  sendDataAction.ReceiveWindowLength = 0; /* RX timeout in us */
360  sendDataAction.data = txDataBuffer; /* Buffer for ACK packet reception */
361  sendDataAction.next_true = &waitAckAction; /* Reschedule the RX */
362  sendDataAction.next_false = &waitAckAction; /* Reschedule the RX */
363  sendDataAction.condRoutine = sendDataCB; /* Condition routine */
364  sendDataAction.dataRoutine = dataRoutine; /* Data routine */
365 
366  waitAckAction.StateMachineNo = STATE_MACHINE_0;
367  waitAckAction.ActionTag = RELATIVE | TIMER_WAKEUP;
368  waitAckAction.WakeupTime = 30+WKP_OFFSET; /* 30 us minimum of internal delay before start action */
369  waitAckAction.ReceiveWindowLength = RX_TIMEOUT_ACK; /* RX timeout in us */
370  waitAckAction.data = receiveBuffer; /* Buffer for ACK packet reception */
371  waitAckAction.next_true = &waitRequestAction; /* Reschedule the RX */
372  waitAckAction.next_false = &sendDataAction; /* Reschedule the RX */
373  waitAckAction.condRoutine = waitAckCB; /* Condition routine */
374  waitAckAction.dataRoutine = dataRoutine; /* Data routine */
375 
376  /* Channel map configuration */
377  uint8_t map[5]= {0xFF,0xFF,0xFF,0xFF,0xFF};
378  RADIO_SetChannelMap(STATE_MACHINE_0, &map[0]);
379 
380  /* Set the channel */
381  RADIO_SetChannel(STATE_MACHINE_0, APP_CHANNEL, 0);
382 
383  /* Sets of the NetworkID and the CRC */
384  RADIO_SetTxAttributes(STATE_MACHINE_0, BLE_NETWORK_ID, 0x555555);
385

387  RADIO_SetReservedArea(&waitRequestAction);
388  RADIO_SetReservedArea(&sendDataAction);
389  RADIO_SetReservedArea(&waitAckAction);
390 }
