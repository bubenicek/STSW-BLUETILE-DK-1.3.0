285 uint8_t RxCallback(ActionPacket* p, ActionPacket* next)
286 {
287  if( (p->status & IRQ_DONE) != 0) {
288 
289  /* received a packet */
290  if( (p->status & BIT_TX_MODE) == 0) {
291 
292  if((p->status & IRQ_RCV_OK) != 0) {
293  /* Check if there is enough space in the FIFO to store the incoming data */
294  if(MAX_PACKET_LENGTH*2 - fifo_size(&blueRec_fifo) > (p->data[1]+ 4 + 4 + 4)){
295  fifo_put_var_len_item(&blueRec_fifo, p->data[1]+2, p->data);
296  fifo_put(&blueRec_fifo, 4, (uint8_t*)(&p->rssi));
297  fifo_put(&blueRec_fifo, 4, (uint8_t*)(&p->timestamp_receive));
298  }
299  HAL_RADIO_ReceivePacket(channel, delay, receivedData, timeOut, RxCallback);
300  }
301  else if((p->status & IRQ_TIMEOUT) != 0) {
302  HAL_RADIO_ReceivePacket(channel,delay, receivedData, timeOut, RxCallback);
303  }
304  else if((p->status & IRQ_CRC_ERR) != 0) {
305  HAL_RADIO_ReceivePacket(channel, delay, receivedData, timeOut, RxCallback);
306  }
307  }
308 
309  /* Transmit complete */
310  else {
311  }
312  }
313  return TRUE;
314 }
