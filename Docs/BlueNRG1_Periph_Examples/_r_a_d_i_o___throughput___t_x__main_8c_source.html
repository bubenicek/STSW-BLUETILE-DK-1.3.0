282 uint8_t TxCallback(ActionPacket* p, ActionPacket* next)
283 {
284  /* received a packet */
285  if((p->status & BIT_TX_MODE) == 0) {
286 
287  if((p->status & IRQ_RCV_OK) != 0) {
288  time2 = MFT1->TNCNT1;
289  timer_reload_store = timer_reload;
290  packet_counter++;
291  }
292  else if((p->status & IRQ_TIMEOUT) != 0) {
293  }
294  else if((p->status & IRQ_CRC_ERR) != 0) {
295  }
296 
297  if(packet_counter != MAX_NUM_PACKET) {
298  sendNewPacket = TRUE;
299  }
300  else
301  {
302  diff = (0xFFFF - time2) + timer_reload_store*65000;
303  time_cumulate += diff;
304  }
305  }
306  /* Transmit complete */
307  else {
308 #ifdef UNIDIRECTIONAL_TEST
309  time2 = MFT1->TNCNT1;
310  timer_reload_store = timer_reload;
311  packet_counter++;
312  if(packet_counter != MAX_NUM_PACKET) {
313  sendNewPacket = TRUE;
314  }
315 #endif
316  }
317  return TRUE;
318 }

375 
376  if(packet_counter == MAX_NUM_PACKET) {
377  if(time_cumulate!=0) {
378  float avg = ((float)time_cumulate)/1000.0/packet_counter;
379  float tput = (((float)packet_counter*DATA_PACKET_LEN*8)*1000.0)/time_cumulate;
380  printf("%d TX packets. %d PCKT LEN. Average time: %d.%03d ms. Data throughput: %d.%03d kbps.\r\n", packet_counter, DATA_PACKET_LEN, PRINT_INT(avg), PRINT_FLOAT(avg), PRINT_INT(tput), PRINT_FLOAT(tput));
381  }

385  sendData[6] = 0;

389  }
390 
391 
392  if(sendNewPacket == TRUE) {
393  sendNewPacket = FALSE;
394  if(packet_counter != 0){
395  diff = (0xFFFF - time2) + timer_reload_store*65000;
396  }
397  time_cumulate += diff;
398  timer_reload = 0;
399  MFT1->TNCNT1 = 0xFFFF;
400 #ifdef UNIDIRECTIONAL_TEST
401  ret = HAL_RADIO_SendPacket(channel, TX_WAKEUP_TIME, sendData, TxCallback);
402 #else
403  ret = HAL_RADIO_SendPacketWithAck(channel, TX_WAKEUP_TIME, sendData, receivedData, RX_TIMEOUT_ACK, TxCallback);
404 #endif
405  if(ret != SUCCESS_0) {

407  }
408  }
409  }
