313 void processCommand(void)
314 {
315  uint8_t command;
316 
317  /* Infinite loop */
318  while(1) {
319 
320  /* Loop until the UART Receive Data Register is not empty */
321  while (UART_GetFlagStatus(UART_FLAG_RXFE) == SET);
322 
323  /* Store the received byte in RxBuffer */
324  command = UART_ReceiveData();
325 
326  /* Parse the command */
327  switch (command) {
328  case TOGGLE_DL1:
329  {
330  /* SPI Master communication */
331  DMASpi_Sending((uint32_t)&command, 1);
332  printf("Toggle LED DL1\r\n");
333  }
334  break;
335  case TOGGLE_DL2:
336  {
337  /* SPI Master communication */
338  DMASpi_Sending((uint32_t)&command, 1);
339  printf("Toggle LED DL2\r\n");
340  }
341  break;
342  case TOGGLE_DL3:
343  {
344  /* SPI Master communication */
345  DMASpi_Sending((uint32_t)&command, 1);
346  printf("Toggle LED DL3\r\n");
347  }
348  break;
349  case TOGGLE_ALL_LED:
350  {
351  uint8_t led[] = {TOGGLE_DL1, TOGGLE_DL2, TOGGLE_DL3};
352  /* SPI Master communication */
353  DMASpi_Sending((uint32_t)led, sizeof(led));
354  printf("All LED Toggled\r\n");
355  }
356  break;
357  default:
358  helpMessage();
359  break;
360  }
361  }
362 }
