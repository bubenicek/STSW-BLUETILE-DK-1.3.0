304 void processCommand(void)
305 {
306  uint8_t command;
307 
308  /* Infinite loop */
309  while(1) {
310 
311  /* Loop until the UART Receive Data Register is not empty */
312  while (UART_GetFlagStatus(UART_FLAG_RXFE) == SET);
313 
314  /* Store the received byte in RxBuffer */
315  command = UART_ReceiveData();
316 
317  /* Parse the command */
318  switch (command) {
319  case TOGGLE_DL1:
320  {
321  /* SPI Master communication */
322  SPI_SendReceiveData(&command, 1);
323  printf("Toggle LED DL1\r\n");
324  }
325  break;
326  case TOGGLE_DL2:
327  {
328  /* SPI Master communication */
329  SPI_SendReceiveData(&command, 1);
330  printf("Toggle LED DL2\r\n");
331  }
332  break;
333  case TOGGLE_DL3:
334  {
335  /* SPI Master communication */
336  SPI_SendReceiveData(&command, 1);
337  printf("Toggle LED DL3\r\n");
338  }
339  break;
340  case TOGGLE_ALL_LED:
341  {
342  uint8_t led[] = {TOGGLE_DL1, TOGGLE_DL2, TOGGLE_DL3};
343  /* SPI Master communication */
344  SPI_SendReceiveData(led, sizeof(led));
345  printf("All LED Toggled\r\n");
346  }
347  break;
348  default:
349  helpMessage();
350  break;
351  }
352  }

421 void SPI_SendReceiveData(uint8_t *command, uint8_t len)
422 {
423  GPIO_ResetBits(SPI_CS_MS_DEMO_PIN);
424 
425  for (uint8_t i=0; i< len; i++) {
426  /* Write data to send to TX FIFO */
427  while(RESET == SPI_GetFlagStatus(SPI_FLAG_TFE));
428  SPI_SendData(command[i]);
429  while(RESET == SPI_GetFlagStatus(SPI_FLAG_RNE));
430  SPI_ReceiveData();
431  while (SET == SPI_GetFlagStatus(SPI_FLAG_BSY));
432  }
433  GPIO_SetBits(SPI_CS_MS_DEMO_PIN);
434 }
