285 void processCommand(void)
286 {
287  uint8_t command;
288 
289  /* Infinite loop */
290  while(1) {
291 
292  /* Get the LEDS actual status */
293  spi_buffer_tx[0] = ((uint8_t)SdkEvalLedGetState(LED3))<<2 | ((uint8_t)SdkEvalLedGetState(LED2))<<1 | ((uint8_t)SdkEvalLedGetState(LED1));
294 
295  /* Prepare buffer to sent through SPI DMA TX channel */
296  DMASpi_Sending((uint32_t)&spi_buffer_tx[0], 1);
297 
298  while(buffer_spi_used==0);
299 
300  /* Parse the command */
301  command = spi_buffer_rx[buffer_spi_tail];
302 
303  if(command == TOGGLE_DL1) {
304  SdkEvalLedToggle(LED1);
305  }
306  else if(command == TOGGLE_DL2) {
307  SdkEvalLedToggle(LED2);
308  }
309  else if(command == TOGGLE_DL3) {
310  SdkEvalLedToggle(LED3);
311  }
312 
313  buffer_spi_tail = (buffer_spi_tail+1)%SPI_BUFF_SIZE;
314  buffer_spi_used--;
315 
316  }
317 }
