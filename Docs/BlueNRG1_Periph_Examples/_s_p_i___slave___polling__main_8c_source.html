276 void processCommand(void)
277 {
278  uint8_t command;
279 
280  /* Infinite loop */
281  while(1) {
282 
283  /* Loop until the UART Receive Data Register is not empty */
284  while (SPI_GetFlagStatus(SPI_FLAG_RNE) == RESET);
285 
286  /* Read command received from SPI master */
287  command = SPI_ReceiveData();
288 
289  /* Parse the command */
290  if(command == TOGGLE_DL1) {
291  SdkEvalLedToggle(LED1);
292  }
293  else if(command == TOGGLE_DL2) {
294  SdkEvalLedToggle(LED2);
295  }
296  else if(command == TOGGLE_DL3) {
297  SdkEvalLedToggle(LED3);
298  }
299 
300  }
301 }

309 void SPI_Slave_Configuration(void)
310 {
311  SPI_InitType SPI_InitStructure;
312  GPIO_InitType GPIO_InitStructure;
313 
314  /* Enable SPI and GPIO clocks */
315  SysCtrl_PeripheralClockCmd(CLOCK_PERIPH_GPIO | CLOCK_PERIPH_SPI, ENABLE);
316 
317  /* Configure SPI pins */
318  GPIO_StructInit(&GPIO_InitStructure);
319  GPIO_InitStructure.GPIO_Pin = SDK_EVAL_SPI_PERIPH_OUT_PIN;
320  GPIO_InitStructure.GPIO_Mode = SDK_EVAL_SPI_PERIPH_OUT_MODE;
321  GPIO_InitStructure.GPIO_Pull = ENABLE;
322  GPIO_InitStructure.GPIO_HighPwr = DISABLE;
323  GPIO_Init(&GPIO_InitStructure);
324 
325  GPIO_InitStructure.GPIO_Pin = SDK_EVAL_SPI_PERIPH_IN_PIN;
326  GPIO_InitStructure.GPIO_Mode = SDK_EVAL_SPI_PERIPH_IN_MODE;
327  GPIO_Init(&GPIO_InitStructure);
328 
329  GPIO_InitStructure.GPIO_Pin = SDK_EVAL_SPI_PERIPH_SCLK_PIN;
330  GPIO_InitStructure.GPIO_Mode = SDK_EVAL_SPI_PERIPH_SCLK_MODE;
331  GPIO_Init(&GPIO_InitStructure);
332 
333  GPIO_InitStructure.GPIO_Pin = SPI_CS_SLAVE_DEMO_PIN;
334  GPIO_InitStructure.GPIO_Mode = SPI_CS_SLAVE_DEMO_MODE;
335  GPIO_Init(&GPIO_InitStructure);
336 
337  /* Configure SPI in master mode */
338  SPI_StructInit(&SPI_InitStructure);
339  SPI_InitStructure.SPI_Mode = SPI_Mode_Slave;

344  SPI_Init(&SPI_InitStructure);
