341 void DMAUart_Sending(uint8_t size)
342 {
343  while(uart_tx_dma_eot==RESET);
344 
345  uart_tx_dma_eot = RESET;
346 
347  DMA_CH_UART_TX->CMAR = (uint32_t)&buffer_uart[buffer_uart_tail];
348 
349  DMA_CH_UART_TX->CNDTR = size;
350  buffer_uart_tail = (buffer_uart_tail+buffer_uart_used)%UART_BUFFER_SIZE;
351 
352  /* DMA_CH_UART_TX enable */
353  DMA_CH_UART_TX->CCR_b.EN = SET;
354 
355 }
