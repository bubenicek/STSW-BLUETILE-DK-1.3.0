659 /* Wait the Low speed oscillator stabilization */
660 void Wait_LS_Stabilization(void)
661 {
662  uint8_t ls_state;
663  uint16_t number_iteration;
664  volatile uint32_t period1;
665  uint32_t period2, tick_result;
666 
667  number_iteration = 0;
668  ls_state = WAIT_FREQ_MIN_STATE;
669  CKGEN_BLE->CLK32K_COUNT = SLOW_COUNT;
670 
671  while ((ls_state != LS_STABILIZATION_ENDED) && (number_iteration <= MAX_NUMBER_ITERATION)) {
672 
673  /* Stabilization State machine */
674  switch(ls_state)
675  {
676  case WAIT_FREQ_MIN_STATE:
677  {
678  /* Wait until the LS freq is minor of 20 KHz */
679  CKGEN_BLE->CLK32K_IT = 1;
680  CKGEN_BLE->CLK32K_PERIOD = 0;
681  while(CKGEN_BLE->CLK32K_IT == 0);
682  period1 = CKGEN_BLE->CLK32K_PERIOD;
683  if (period1 <= MIN_CLOCK_PERIOD) {
684  ls_state = WAIT_FREQ_STABILIZATION_STATE;
685  }
686  }
687  break;
688  case WAIT_FREQ_STABILIZATION_STATE:
689  {
690  CKGEN_BLE->CLK32K_IT = 1;
691  CKGEN_BLE->CLK32K_PERIOD = 0;
692  while(CKGEN_BLE->CLK32K_IT == 0);
693  period2 = CKGEN_BLE->CLK32K_PERIOD;
694  if (period2 > period1)
695  tick_result = period2 - period1;
696  else
697  tick_result = period1 - period2;
698  if (tick_result <= (period1>>10)) {
699  ls_state = LS_STABILIZATION_ENDED;
700  }
701  period1 = period2;
702  }
703  break;
704  default:
705  /* Nothing to do */
706  break;
707  }
708  number_iteration++;
709  }
710 }

775 /* Function to evaluate if the application has enough time to load the hot table */
776 __STATIC_INLINE void nextRF_transaction(uint16_t reset_reason)
777 {
778  uint32_t currTimeTHR;
779  int32_t time_diff = -1;
780  PartInfoType Device_partInfo;
781  uint8_t load_hot_table = 1;
782 
783  /* Wakeup from radio transaction. So, nothing to do */
784  if (reset_reason & RESET_BLE_WAKEUP_FROM_TIMER1) {
785  return;
786  }
787 
788  /* Verify device version */
789  HAL_GetPartInfo(&Device_partInfo);
790  if (((Device_partInfo.die_major<<4)|(Device_partInfo.die_cut)) < WA_DEVICE_VERSION) {
791  load_hot_table = 0;
792  }
793 
794  /* Verify if the BLE radio is NOT active */
795  if((*(volatile uint32_t*)0x200000C0 & 8U) == 0U) {
796  if (load_hot_table)
797  SET_BORconfigStatus(TRUE);
798  return;
799  }
800 
801  /* Check if the next transaction is enough in the future */
802  currTimeTHR = BLUE_CTRL->CURRENT_TIME + TIMEOUT_THRESHOLD;
803  time_diff = TIME24_DIFF(rfTimeout, currTimeTHR);
804  if (time_diff > 0) {
805  /* There is enough time to load the hot table */
806  if (load_hot_table)
807  SET_BORconfigStatus(TRUE);
808  /* Restore the Timeout register content after wakeup */
809  BLUE_CTRL->TIMEOUT = rfTimeout;
810  } else {
811  /* Nothing to do, too close to RF timeout. Wait the radio transaction */
812  }
813 
814 }

927  CKGEN_BLE->CLK32K_PERIOD = 0;
